-- FunciÃ³n: calculate_user_points
CREATE OR REPLACE FUNCTION calculate_user_points(
    p_position INTEGER,
    p_total_participants INTEGER,
    p_error_reduction DECIMAL,
    p_streak INTEGER
) RETURNS INTEGER AS $$
DECLARE
    v_base_points INTEGER := 5;
    v_performance_points INTEGER;
    v_improvement_bonus INTEGER := 0;
    v_multiplier DECIMAL := 1.0;
    v_total INTEGER;
BEGIN
    -- Calcular puntos por rendimiento
    v_performance_points := CASE
        WHEN p_position = 1 THEN 20
        WHEN p_position = 2 THEN 19
        WHEN p_position = 3 THEN 18
        WHEN p_position <= p_total_participants * 0.1 THEN 15 + FLOOR(3 * (1 - (p_position - 3) / (p_total_participants * 0.1 - 3)))
        WHEN p_position <= p_total_participants * 0.25 THEN 12 + FLOOR(3 * (1 - (p_position - p_total_participants * 0.1) / (p_total_participants * 0.15)))
        WHEN p_position <= p_total_participants * 0.5 THEN 8 + FLOOR(4 * (1 - (p_position - p_total_participants * 0.25) / (p_total_participants * 0.25)))
        ELSE FLOOR(7 * (1 - (p_position - p_total_participants * 0.5) / (p_total_participants * 0.5)))
    END;
    
    -- Bonus por mejora
    IF p_error_reduction >= 5 THEN
        v_improvement_bonus := 5;
    END IF;
    
    -- Multiplicador por racha
    IF p_streak >= 16 THEN
        v_multiplier := 2.0;
    ELSIF p_streak >= 8 THEN
        v_multiplier := 1.5;
    END IF;
    
    -- Calcular total con cap de 25
    v_total := LEAST(25, FLOOR((v_base_points + v_performance_points + v_improvement_bonus) * v_multiplier));
    
    RETURN v_total;
END;
$$ LANGUAGE plpgsql;
-- Trigger: update_user_elo
CREATE OR REPLACE FUNCTION update_user_elo() RETURNS TRIGGER AS $$
BEGIN
    -- Actualizar ELO del usuario
    UPDATE users 
    SET current_elo = current_elo + NEW.total_points
    WHERE id = NEW.user_id;
    
    -- Registrar en historial
    INSERT INTO elo_history (user_id, week_number, elo_before, elo_after, change)
    SELECT 
        NEW.user_id,
        (SELECT week_number FROM weekly_simulations WHERE id = NEW.simulation_id),
        current_elo - NEW.total_points,
        current_elo,
        NEW.total_points
    FROM users WHERE id = NEW.user_id;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trigger_update_elo
AFTER INSERT ON user_results
FOR EACH ROW EXECUTE FUNCTION update_user_elo();
